webpack uses a configuration file to define how it should process and bundle your application. This section provides detailed documentation for all configuration options available in webpack.

## Configuration File

webpack expects a configuration file named `webpack.config.js` in the root of your project. This file should export a configuration object:

```javascript
const path = require('path');

module.exports = {
  mode: 'production',
  entry: './src/index.js',
  output: {
    path: path.resolve(__dirname, 'dist'),
    filename: 'bundle.js'
  }
};
```

## Configuration Types

webpack supports multiple configuration formats:

### Object Export

```javascript
module.exports = {
  // configuration
};
```

### Function Export

Receives environment and argv as parameters:

```javascript
module.exports = (env, argv) => {
  return {
    mode: env.production ? 'production' : 'development',
    devtool: env.production ? 'source-map' : 'eval',
    // more options
  };
};
```

### Promise Export

Return a Promise that resolves to a configuration:

```javascript
module.exports = () => {
  return new Promise((resolve) => {
    setTimeout(() => {
      resolve({
        entry: './app.js',
        // more options
      });
    }, 5000);
  });
};
```

### Multiple Configurations

Export an array of configurations:

```javascript
module.exports = [
  {
    entry: './src/app.js',
    output: { filename: 'app.js' }
  },
  {
    entry: './src/vendor.js',
    output: { filename: 'vendor.js' }
  }
];
```

## Mode

<ParamField path="mode" type="'development' | 'production' | 'none'">
  Enable built-in optimizations for the specified environment.
  
  **Default:** `'production'`
  
  - `'development'`: Optimized for build speed and debugging
  - `'production'`: Optimized for runtime performance and smaller bundle size
  - `'none'`: No default optimizations
</ParamField>

## Context

<ParamField path="context" type="string">
  The base directory (absolute path) for resolving entry points and loaders.
  
  **Default:** `process.cwd()` (current working directory)
</ParamField>

```javascript
module.exports = {
  context: path.resolve(__dirname, 'app')
};
```

## Other Top-Level Options

<ParamField path="name" type="string">
  Name of the configuration. Used when loading multiple configurations.
</ParamField>

<ParamField path="bail" type="boolean">
  Report the first error as a hard error instead of tolerating it.
  
  **Default:** `false`
</ParamField>

<ParamField path="cache" type="boolean | object">
  Cache generated modules and chunks to improve performance for multiple incremental builds.
  
  **Default:** `false` in production, `{ type: 'memory' }` in development
</ParamField>

<ParamField path="dependencies" type="string[]">
  References to other configurations to depend on.
</ParamField>

<ParamField path="profile" type="boolean">
  Capture timing information for each module.
  
  **Default:** `false`
</ParamField>

<ParamField path="parallelism" type="number">
  The number of parallel processed modules in the compilation.
  
  **Default:** `100`
</ParamField>

## Example Configuration

```javascript
const path = require('path');

module.exports = {
  mode: 'production',
  context: path.resolve(__dirname, 'src'),
  entry: './index.js',
  output: {
    path: path.resolve(__dirname, 'dist'),
    filename: '[name].[contenthash].js'
  },
  module: {
    rules: [
      {
        test: /\.js$/,
        exclude: /node_modules/,
        use: 'babel-loader'
      }
    ]
  },
  optimization: {
    splitChunks: {
      chunks: 'all'
    }
  }
};
```

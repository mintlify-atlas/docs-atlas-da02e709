title: Stats
description: Configure compilation output statistics and logging in webpack
---

The `stats` configuration controls what bundle information gets displayed during compilation. You can customize the output to show exactly what you need.

## Basic Configuration

```javascript
module.exports = {
  stats: 'errors-only'
};
```

## Presets

<ParamField path="stats" type="string | boolean | object">
  Control compilation output statistics.
  
  **Default:** Varies by webpack configuration
  
  ```javascript
  // String preset
  stats: 'minimal'
  
  // Boolean
  stats: false  // Disable all output
  
  // Object for fine control
  stats: {
    colors: true,
    modules: false
  }
  ```
</ParamField>

### Available Presets

<ParamField path="'none'">
  Output nothing.
  
  ```javascript
  stats: 'none'
  ```
</ParamField>

<ParamField path="'errors-only'">
  Only output errors.
  
  ```javascript
  stats: 'errors-only'
  ```
  
  Useful for CI/CD where you only care about failures.
</ParamField>

<ParamField path="'errors-warnings'">
  Only output errors and warnings.
  
  ```javascript
  stats: 'errors-warnings'
  ```
  
  Recommended for production builds.
</ParamField>

<ParamField path="'minimal'">
  Output only when errors or new compilation happens.
  
  ```javascript
  stats: 'minimal'
  ```
  
  Good for watch mode development.
</ParamField>

<ParamField path="'normal'">
  Standard output (default).
  
  ```javascript
  stats: 'normal'
  ```
</ParamField>

<ParamField path="'detailed'">
  Output everything except chunkModules and chunkRootModules.
  
  ```javascript
  stats: 'detailed'
  ```
</ParamField>

<ParamField path="'verbose'">
  Output everything.
  
  ```javascript
  stats: 'verbose'
  ```
  
  Useful for debugging builds.
</ParamField>

<ParamField path="'summary'">
  Output webpack version, warnings count, and errors count.
  
  ```javascript
  stats: 'summary'
  ```
</ParamField>

## Stats Options

For fine-grained control, use an object:

```javascript
module.exports = {
  stats: {
    // Options
  }
};
```

### Display Options

<ParamField path="stats.colors" type="boolean">
  Enable colored output.
  
  **Default:** Auto-detected (true if terminal supports colors)
  
  ```javascript
  stats: {
    colors: true
  }
  ```
</ParamField>

<ParamField path="stats.hash" type="boolean">
  Display the compilation hash.
  
  **Default:** `true`
  
  ```javascript
  stats: {
    hash: true
  }
  ```
</ParamField>

<ParamField path="stats.version" type="boolean">
  Display webpack version.
  
  **Default:** `true`
  
  ```javascript
  stats: {
    version: true
  }
  ```
</ParamField>

<ParamField path="stats.timings" type="boolean">
  Display compilation timing.
  
  **Default:** `true`
  
  ```javascript
  stats: {
    timings: true
  }
  ```
</ParamField>

<ParamField path="stats.builtAt" type="boolean">
  Display build date and time.
  
  **Default:** `true`
  
  ```javascript
  stats: {
    builtAt: true
  }
  ```
</ParamField>

### Assets

<ParamField path="stats.assets" type="boolean">
  Display asset information.
  
  **Default:** `true`
  
  ```javascript
  stats: {
    assets: true,
    assetsSpace: 15,  // Limit to 15 assets
    assetsSort: 'size'  // Sort by size
  }
  ```
</ParamField>

<ParamField path="stats.assetsSort" type="string">
  Sort assets by field.
  
  Options: `'id'`, `'name'`, `'size'`, `'chunks'`, `'errors'`, `'warnings'`, `'failed'`, `'emitted'`
  
  **Default:** `'id'`
</ParamField>

<ParamField path="stats.assetsSpace" type="number">
  Limit the number of assets displayed.
  
  ```javascript
  stats: {
    assetsSpace: 10  // Show only 10 assets
  }
  ```
</ParamField>

<ParamField path="stats.excludeAssets" type="string | RegExp | function | array">
  Exclude assets from stats output.
  
  ```javascript
  stats: {
    excludeAssets: [
      /\.map$/,
      /\.json$/
    ]
  }
  ```
</ParamField>

<ParamField path="stats.groupAssetsByInfo" type="boolean">
  Group assets by asset info (e.g., immutable, development, hotModuleReplacement).
  
  **Default:** `true`
</ParamField>

<ParamField path="stats.groupAssetsByPath" type="boolean">
  Group assets by path.
  
  **Default:** `true`
</ParamField>

<ParamField path="stats.groupAssetsByExtension" type="boolean">
  Group assets by extension.
  
  **Default:** `true`
</ParamField>

<ParamField path="stats.groupAssetsByChunk" type="boolean">
  Group assets by chunk.
  
  **Default:** `true`
</ParamField>

<ParamField path="stats.groupAssetsByEmitStatus" type="boolean">
  Group assets by emit status.
  
  **Default:** `true`
</ParamField>

## Programmatic Usage

```javascript
const webpack = require('webpack');
const config = require('./webpack.config.js');

const compiler = webpack(config);

compiler.run((err, stats) => {
  if (err) {
    console.error(err);
    return;
  }
  
  console.log(stats.toString({
    colors: true,
    modules: false,
    children: false,
    chunks: false,
    chunkModules: false
  }));
});
```

## Stats JSON

Generate a JSON file with stats:

```bash
webpack --json > stats.json
```

Or programmatically:

```javascript
const fs = require('fs');
const webpack = require('webpack');

webpack(config, (err, stats) => {
  if (err) throw err;
  
  fs.writeFileSync(
    'stats.json',
    JSON.stringify(stats.toJson())
  );
});
```

Analyze with:
- [webpack-bundle-analyzer](https://github.com/webpack-contrib/webpack-bundle-analyzer)
- [webpack.github.io/analyse](https://webpack.github.io/analyse/)
- [Statoscope](https://github.com/statoscope/statoscope)

## Environment-Based Configuration

```javascript
module.exports = (env, argv) => {
  const isDevelopment = argv.mode === 'development';
  const isCI = process.env.CI === 'true';
  
  return {
    stats: isDevelopment
      ? 'minimal'
      : isCI
        ? {
            colors: false,
            errors: true,
            warnings: true,
            errorDetails: true
          }
        : 'errors-warnings'
  };
};
```

## Custom Stats Reporter

```javascript
class CustomStatsPlugin {
  apply(compiler) {
    compiler.hooks.done.tap('CustomStatsPlugin', (stats) => {
      const info = stats.toJson({
        all: false,
        errors: true,
        warnings: true,
        assets: true
      });
      
      console.log('\n=== Build Complete ===');
      console.log(`Time: ${info.time}ms`);
      console.log(`Hash: ${info.hash}`);
      console.log(`Assets: ${info.assets.length}`);
      
      if (info.errors.length > 0) {
        console.log(`\nErrors: ${info.errors.length}`);
        info.errors.forEach(err => console.error(err));
      }
      
      if (info.warnings.length > 0) {
        console.log(`\nWarnings: ${info.warnings.length}`);
      }
    });
  }
}

module.exports = {
  plugins: [new CustomStatsPlugin()]
};
```

## Performance Tips

1. **Disable unused stats in development**:
   ```javascript
   stats: {
     modules: false,
     children: false
   }
   ```

2. **Use presets for simplicity**:
   ```javascript
   stats: 'minimal'
   ```

3. **Filter large outputs**:
   ```javascript
   stats: {
     excludeModules: /node_modules/,
     modulesSpace: 15
   }
   ```

4. **CI-specific configuration**:
   ```javascript
   stats: {
     colors: false,
     modules: false,
     entrypoints: false
   }
   ```
